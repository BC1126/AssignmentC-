INSERT INTO Seats (HallId, SeatIdentifier, IsPremium, IsWheelchair, IsActive)
SELECT 
    h.HallId,
    r.RowLetter + CAST(s.Number AS VARCHAR(2)) AS SeatIdentifier,
    CASE WHEN r.RowLetter IN ('D', 'E') THEN 1 ELSE 0 END AS IsPremium,
    CASE WHEN r.RowLetter = 'A' AND s.Number <= 2 THEN 1 ELSE 0 END AS IsWheelchair,
    1 AS IsActive
FROM Halls h
CROSS JOIN (
    -- Define your Rows here
    SELECT 'A' AS RowLetter UNION ALL SELECT 'B' UNION ALL 
    SELECT 'C' UNION ALL SELECT 'D' UNION ALL SELECT 'E'
) r
CROSS JOIN (
    -- Define your Seat Numbers here
    SELECT 1 AS Number UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL 
    SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL 
    SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10
) s
-- This prevents duplicates if you run the script twice
WHERE NOT EXISTS (
    SELECT 1 FROM Seats existing 
    WHERE existing.HallId = h.HallId 
    AND existing.SeatIdentifier = r.RowLetter + CAST(s.Number AS VARCHAR(2))
);
